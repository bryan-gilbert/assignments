Question 3: Series Searcher

Select your options as per the assignment document and see the results.

Here is the main calculation function. This code assumes the caller has completed checks on the inputs before calling this function.

{
	let start = inputs.seriesStart;
	let end = inputs.seriesEnd;
	let incr = inputs.seriesIncrement;
	let type = inputs.seriesType;
	  // convert number to regexp
	let specifiedDigit = ''+ inputs.specifiedDigit;
	var re = new RegExp(specifiedDigit, 'g');
	  // the results:
	let series = [];
	let total = 0;
	for(var i = start; i < end; i += incr) {
		let str = '' + i;
		var count = (str.match(re) || []).length;
		if(type === 1) {
			series.push(i);
			total += count;
		} else if(type === 2) {
			if(i % 2 === 0) {
				series.push(i);
				total += count;
			}
		} else if(type === 3) {
			if(i % 2 === 1) {
				series.push(i);
				total += count;
			}
		}
	}
	inputs.calculatedResults = total;
	inputs.series = series;
}
